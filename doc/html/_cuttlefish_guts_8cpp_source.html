00027 // Interrupts:
00028 // Timer2 has highest priority, but you can't have a flexible interrupt rate as well as pwm.
00029 // The TimerOne library is more flexible and allows both pwm and interrupts,
00030 // by using ICR1 for interrupt rate and OCR1A (pin 9) (or OCR1B pin 10) for pwm level
00031 // So we use Timer1 for audio.
00032 // Ideally we'd use Timer0 for control because it has lowest priority, but Arduino uses
00033 // it for time and delay functions, so we use Timer 2 for control and hope it won't
00034 // interfere with the audio interrupt.

00048 void startCuttlefish(unsigned int control_rate_hz)
00049 {
00050 
00051                 // audio
00052                 pinMode(AUDIO_CHANNEL_1_PIN, OUTPUT);                           // set pin to output for audio
00053                 Timer1.initialize(1000000/AUDIO_RATE);      // set period
00054                 Timer1.pwm(AUDIO_CHANNEL_1_PIN, 512);       // pwm pin, 50% duty cycle
00055                 Timer1.attachInterrupt(outputAudio);                            // call outputAudio() on each interrupt

00058                 TimerTwo::init(1000000/control_rate_hz,updateControl); // set period, attach updateControl()
00059                 TimerTwo::start();

00069 void audioHook()
00070 {
00071 
00072                 static unsigned char in_pos = 0;
00073                 static unsigned int num_in = 0;
00074                 unsigned int gap;
00075                 ATOMIC_BLOCK(ATOMIC_RESTORESTATE)
00076                 { // because 16bit access to num_out which can be changed in Buffer::out() used in audio interrupt
00077                                 gap = num_in - num_out; // wraps to a big number if it's negative
00078                 }
00079                 if (gap < BUFFER_NUM_TABLE_CELLS)
00080                 {
00081                                 in_pos++;
00082                                 num_in++;
00083                                 bufarray[in_pos] = updateAudio() + AUDIO_BIAS;
00084                 }
00085 
00086 }
