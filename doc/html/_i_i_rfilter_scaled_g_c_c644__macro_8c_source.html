00068 int IIR2(int xx)
00069 {
00070         int yy  ;
00071         // sum the 5 terms: yy += xx*coeff
00072         // and update the state variables
00073         // as soon as possible
00074         yy = multfix(xn_2,b3);
00075         xn_2 = xn_1;
00076         yy = yy + multfix(xn_1,b2);
00077         xn_1 = xx;
00078         yy = yy + multfix(xx,b1);
00079         yy = yy + multfix(yn_2,a3);
00080         yn_2 = yn_1;
00081         yy = yy + multfix(yn_1,a2);
00082         yy = yy>>scaleFactorShift ; // divide by 16 = a(1)
00083         yn_1 = yy;
00084         return yy;
00085 }
