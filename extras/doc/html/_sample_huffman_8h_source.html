22  * The header file contains two lengthy arrays:
23  * One is "sounddata" which must fit into Flash RAM (available in total: 16k for ATMega168, 32k for ATMega328)

42 // Get one bit from sound data
43 inline int getbit(unsigned long pos,boolean autoloadonbit0=false)
44 {
45  const int b = pos&7;
46  static unsigned char bt;
47  if(!autoloadonbit0 || !b)
48  // read indexed byte from Flash memory
49  bt = pgm_read_byte(&sounddata[pos>>3]);
50  // extract the indexed bit
51  return (bt>>(7-b))&1;
52 }

54 // Decode bit stream using Huffman codes
55 static int decode(unsigned long &pos,int const *huffcode)
56 {
57  unsigned long p = pos;
58  do {
59  const int b = getbit(p++,true);
60  if(b) {
61  const int offs = *huffcode;
62  huffcode += offs?offs+1:2;
63  }
64  } while(*(huffcode++));
65  pos = p;
66  return *huffcode;
67 }
